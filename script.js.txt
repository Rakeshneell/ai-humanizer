async function humanizeText() {
    const text = document.getElementById('inputText').value;
    const output = document.getElementById('outputText');
    const apiKey = "YOUR_GEMINI_API_KEY_HERE"; // এখানে আপনার কি বসান

    if (!text) return alert("দয়া করে কিছু লিখুন!");

    output.value = "প্রসেসিং হচ্ছে... একটু অপেক্ষা করুন।";

    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            contents: [{
                parts: [{ text: `Rewrite this text to sound exactly like a human wrote it. Remove robotic patterns, vary sentence length, and make it natural: ${text}` }]
            }]
        })
    });

    const data = await response.json();
    output.value = data.candidates[0].content.parts[0].text;
}
